43 void lbann_callback_print::on_epoch_end(Model* m) {
44  lbann_comm* comm = m->get_comm();
45  if (comm->am_model_master()) {
46  DataType train_acc = m->get_train_accuracy();
47  DataType validate_acc = m->get_validate_accuracy();
48  if (comm->am_world_master()) {
49  std::vector<DataType> train_accs(comm->get_num_models());
50  std::vector<DataType> validate_accs(comm->get_num_models());
51  comm->intermodel_gather(train_acc, train_accs);
52  comm->intermodel_gather(validate_acc, validate_accs);
53  for (size_t i = 0; i < train_accs.size(); ++i) {
54  std::cout << "Model " << i;
55  std::cout << " @" << m->get_cur_step() << " steps";
56  std::cout << " Training accuracy: " << train_accs[i] << "%";
57  std::cout << " Validation accuracy: " << validate_accs[i] << "%";
58  std::cout << std::endl;
59  }
60  } else {
61  comm->intermodel_gather(train_acc, comm->get_intermodel_master());
62  comm->intermodel_gather(validate_acc, comm->get_intermodel_master());
63  }
64  for (Layer* layer : m->get_layers()) {
65  layer->epoch_print();
66  }
67  }
68 }

70 void lbann_callback_print::on_test_end(Model* m) {
71  lbann_comm* comm = m->get_comm();
72  if (comm->am_model_master()) {
73  DataType test_acc = m->get_test_accuracy();
74  if (comm->am_world_master()) {
75  std::vector<DataType> accs(comm->get_num_models());
76  comm->intermodel_gather(test_acc, accs);
77  for (size_t i = 0; i < accs.size(); ++i) {
78  std::cout << "Model " << i << " external validation accuracy: ";
79  std::cout << accs[i] << "%" << std::endl;
80  }
81  } else {
82  comm->intermodel_gather(test_acc, comm->get_intermodel_master());
83  }
84  }
