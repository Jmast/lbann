36 lbann::ConvolutionalLayer::ConvolutionalLayer(const uint index,
37  const int _ConvDim,
38  const int _InputChannels,
39  const int* _InputDims,
40  const int _OutputChannels,
41  const int* _FilterDims,
42  const uint miniBatchSize,
43  lbann_comm* comm,
44  Optimizer* optimizer)
45  : Layer(index, comm, optimizer, miniBatchSize),
46  ConvDim(_ConvDim), InputChannels(_InputChannels),
47  OutputChannels(_OutputChannels)
48 {
49  // Initialize input, output, and filter dimensions
50  for(int d=0; d<ConvDim; d++) {
51  InputDims[d] = _InputDims[d];
52  FilterDims[d] = _FilterDims[d];
53  OutputDims[d] = InputDims[d] - FilterDims[d] + 1;
54  }
55  for(int d=ConvDim; d<3; d++) {
56  InputDims[d] = 1;
57  FilterDims[d] = 1;
58  OutputDims[d] = 1;
59  }
60 
61  // Matrices should be in Star,Star and Star,VC distributions
62  delete WB;
63  delete WB_D;
64  delete Zs;
65  delete Ds;
66  delete Ds_Temp;
67  delete Acts;
68  WB = new StarMat(comm->get_model_grid());
69  WB_D = new StarMat(comm->get_model_grid());
70  Zs = new StarVCMat(comm->get_model_grid());
71  Ds = new StarVCMat(comm->get_model_grid());
72  Ds_Temp = new StarVCMat(comm->get_model_grid());
73  Acts = new StarVCMat(comm->get_model_grid());
74 
75  // TODO: obtain dimensions from cudnnNet
76  NumNeurons = OutputChannels*OutputDims[0]*OutputDims[1]*OutputDims[2];
77  std::cout << "conv layer neurons = "<< NumNeurons << std::endl;
78 
79 #ifdef __LIB_CUDNN
80  cudnnLayer = NULL;
81 #endif
82 }

91 void lbann::ConvolutionalLayer::setup(CudnnNet<DataType> *cudnnNet)
92 {
93 #ifdef __LIB_CUDNN
94 
95  // setup cudnn-based convolutional layer instance
96  int cudnnSrcTensorDims[5] = {1, InputChannels,
97  InputDims[0], InputDims[1], InputDims[2]};
98  int cudnnFilterDims[5] = {OutputChannels, InputChannels,
99  FilterDims[0], FilterDims[1], FilterDims[2]};
100  int cudnnDstTensorDims[5] = {1, OutputChannels,
101  OutputDims[0], OutputDims[1], OutputDims[2]};
102  if (cudnnLayer) delete cudnnLayer;
103  cudnnLayer = new CudnnLayer<DataType>(cudnnNet);
104  cudnnLayer->setupConvLayer(ConvDim, cudnnSrcTensorDims,
105  cudnnFilterDims, cudnnDstTensorDims);
106  //printf("convolutional layer: outputsize: %d\n", cudnnLayer->dstDataSize);
107  //printf("convolutional layer: outputdim: %d %d %d %d\n", OutputDim[0], OutputDim[1], OutputDim[2], OutputDim[3]);
108 
109  if(optimizer != NULL) {
110  optimizer->setup(1, cudnnLayer->filterSize);
111  }
112 
113  // Initialize matrices
114  DataType var_scale = sqrt(3.0 / (cudnnLayer->srcDataSize));
115  Gaussian(*WB, cudnnLayer->filterSize, 1, (DataType)0.0, var_scale);
116  Zeros(*WB_D, cudnnLayer->filterSize, 1);
117  Ones(*Zs, cudnnLayer->dstDataSize+1, m_mini_batch_size);
118  Zeros(*Ds, cudnnLayer->srcDataSize+1, m_mini_batch_size);
119  Zeros(*Ds_Temp, cudnnLayer->srcDataSize+1, m_mini_batch_size);
120  Ones(*Acts, cudnnLayer->dstDataSize+1, m_mini_batch_size);
121 
122 #endif
123 }

125 void lbann::ConvolutionalLayer::fp_linearity(ElMat& _WB, ElMat& _X, ElMat& _Z, ElMat& _Y) {
126 
127  // Convert matrices to desired formats
128  DistMatrixReadProxy<DataType,DataType,STAR,STAR> WBProxy(_WB);
129  DistMatrixReadProxy<DataType,DataType,STAR,VC> XProxy(_X);
130  DistMatrixWriteProxy<DataType,DataType,STAR,VC> ZProxy(_Z);
131  DistMatrixWriteProxy<DataType,DataType,STAR,VC> YProxy(_Y);
132  StarMat& WB = WBProxy.Get();
133  StarVCMat& X = XProxy.Get();
134  StarVCMat& Z = ZProxy.Get();
135  StarVCMat& Y = YProxy.Get();
136 
137  // Get local matrices
138  Mat& XLocal = X.Matrix();
139  Mat& ZLocal = Z.Matrix();
140  Mat& YLocal = Y.Matrix();
141 
142  // Iterate through samples in minibatch
143  for (int j = 0; j < XLocal.Width(); j++) {
144  DataType* src = XLocal.Buffer(0, j);
145  DataType* ftr = WB.Buffer();
146  DataType* dst = ZLocal.Buffer(0, j);
147 
148 #ifdef __LIB_CUDNN
149  // Apply convolution
150  cudnnLayer->convForwardHost(src, ftr, dst);
151 #endif
152  }
153 
154  // Z and Y are identical after fp linearity step
155  Copy(ZLocal, YLocal);
156 
157 }

159 void lbann::ConvolutionalLayer::bp_linearity() {
160 
161  // Convert matrices to desired formats
162  DistMatrixReadProxy<DataType,DataType,STAR,VC> OutputDeltaProxy(*bp_input);
163  DistMatrixReadProxy<DataType,DataType,STAR,VC> InputProxy(*fp_input); // TODO: store from fp step
164  StarVCMat& OutputDelta = OutputDeltaProxy.Get();
165  StarVCMat& Input = InputProxy.Get();
166 
167  // Get local matrices
168  Mat& InputLocal = Input.Matrix();
169  Mat& OutputLocal = Acts->Matrix();
170  Mat& InputDeltaLocal = Ds_Temp->Matrix();
171  Mat& OutputDeltaLocal = OutputDelta.Matrix();
172  Mat FilterDeltaTemp(WB_D->Height(), WB_D->Width());
173 
174  // Clear gradient
175  Zero(*WB_D);
176 
177  // Iterate through samples in minibatch
178  for (int j = 0; j < InputLocal.Width(); j++) {
179  DataType* src = InputLocal.Buffer(0, j);
180  DataType* ftr = WB->Buffer();
181  DataType* dst = OutputLocal.Buffer(0, j);
182  DataType* srcD = InputDeltaLocal.Buffer(0, j);
183  DataType* ftrD = FilterDeltaTemp.Buffer();
184  DataType* dstD = OutputDeltaLocal.Buffer(0, j);
185 
186 #ifdef __LIB_CUDNN
187  cudnnLayer->convBackwardHost(src, ftr, dstD, ftrD, srcD);
188 #endif
189 
190  WB_D->Matrix() += FilterDeltaTemp;
191 
192  }
193 
194  AllReduce(*WB_D, mpi::COMM_WORLD);
195  *WB_D *= 1.0/get_effective_minibatch_size();
196 
197 }
