82  void update_weight_bias_matrix(ElMat& WB_D, ElMat& WB) {
83 
84  // KERAS: lr = self.lr * (1.0 / (1.0 + self.decay * self.iterations))
85  lr = lr * (1.0 / (1.0 + decay * iterations));
86  iterations++;
87 
88  // velocity
89  // KERAS: v = self.momentum * m - lr * g # velocity
90  Scale(momentum, velocity);
91  Scale(lr, WB_D); // step
92  Axpy(-1, WB_D.Matrix(), velocity.Matrix()); // Update // TODO
93 
94  if(nesterov) {
95  //KERAS: new_p = p + self.momentum * v - lr * g
96  Copy(velocity, nesterov_ag);
97  Scale(momentum, nesterov_ag);
98  Axpy(-1, WB_D.Matrix(), nesterov_ag.Matrix()); // TODO
99  Axpy(1., nesterov_ag.Matrix(), WB.Matrix()); // TODO
100  }else {
101  //KERAS: new_p = p + v
102  Axpy(1., velocity.Matrix(), WB.Matrix()); // TODO
103  }
104 
105  }
