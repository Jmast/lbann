67 void lbann::lbann_summary::reduce_stdev(std::string tag, DistMat& mat,
68  int64_t step) {
69  // Compute the local sum and squared sum.
70  DataType sum = 0.0f;
71  DataType sqsum = 0.0f;
72  Mat& local_mat = mat.Matrix();
73  for (int row = 0; row < local_mat.Height(); ++row) {
74  for (int col = 0; col < local_mat.Width(); ++col) {
75  DataType v = local_mat.Get(row, col);
76  sum += v;
77  sqsum += v * v;
78  }
79  }
80  pending_stdevs.emplace_back(tag, step, sum, sqsum, mat.Height() * mat.Width());
81 }

112 void lbann::lbann_summary::flush_means() {
113  for (const auto& op : pending_means) {
114  if (comm->am_model_master()) {
115  DataType model_sum = comm->model_reduce(op.local);
116  DataType model_mean = model_sum / op.num;
117  gather_scalar_summary(op.tag, model_mean, op.step);
118  } else {
119  comm->model_reduce(op.local, comm->get_model_master());
120  }
121  }
122  pending_means.clear();

149 void lbann::lbann_summary::flush_stdevs() {
150  for (const auto& op : pending_stdevs) {
151  // Compute the model sample standard deviation as:
152  // sqrt[1/(n-1) (sqsum - (1/n)*sum^2)]
153  // The n-1 is to use an unbiased variance estimate.
154  // This unrolls the usual formulation of standard deviation some, to avoid
155  // global operations when pushing the operation.
156  if (comm->am_model_master()) {
157  DataType model_sum = comm->model_reduce(op.local);
158  DataType model_sqsum = comm->model_reduce(op.local2);
159  DataType model_stdev = std::sqrt((model_sqsum -
160  model_sum * model_sum / op.num) /
161  (op.num - 1));
162  gather_scalar_summary(op.tag, model_stdev, op.step);
163  } else {
164  comm->model_reduce(op.local, comm->get_model_master());
165  comm->model_reduce(op.local2, comm->get_model_master());
166  }
167  }
168  pending_stdevs.clear();
169 }

192 DataType lbann::lbann_summary::local_sum(DistMat& _mat) {
193  Mat& mat = _mat.Matrix();
194  // Note there are more numerically stable ways to compute a sum.
195  DataType sum = 0.0;
196  for (int row = 0; row < mat.Height(); ++row) {
197  for (int col = 0; col < mat.Width(); ++col) {
198  sum += mat.Get(row, col);
199  }
200  }
201  return sum;
202 }

208 void lbann::lbann_summary::gather_scalar_summary(std::string tag, DataType s,
209  int64_t step) {
210  if (comm->am_world_master()) {
211  std::vector<DataType> data(comm->get_num_models());
212  comm->intermodel_gather(s, data);
213  for (size_t model = 0; model < data.size(); ++model) {
214  sw->add_scalar(prepend_model(tag, model), data[model], step);
215  }
216  } else {
217  comm->intermodel_gather(s, comm->get_intermodel_master());
218  }
219 }
