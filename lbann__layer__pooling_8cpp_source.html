36 lbann::PoolingLayer::PoolingLayer(const uint index,
37  const int _PoolDim,
38  const int _Channels,
39  const int* _InputDims,
40  const int* _PoolWindowDim,
41  const int* _PoolPad,
42  const int* _PoolStride,
43  const int _PoolMode,
44  const uint miniBatchSize,
45  lbann_comm* comm,
46  Optimizer* optimizer)
47  : Layer(index, comm, optimizer, miniBatchSize),
48  PoolDim(_PoolDim), Channels(_Channels), PoolMode(_PoolMode)
49 {
50  for(int d=0; d<PoolDim; d++) {
51  InputDims[d] = _InputDims[d];
52  PoolWindowDim[d] = _PoolWindowDim[d];
53  PoolPad[d] = _PoolPad[d];
54  PoolStride[d] = _PoolStride[d];
55  OutputDims[d] = (InputDims[d]+2*PoolPad[d])/PoolStride[d];
56  }
57  for(int d=PoolDim; d<3; d++) {
58  InputDims[d] = 1;
59  PoolWindowDim[d] = 1;
60  PoolPad[d] = 1;
61  PoolStride[d] = 1;
62  OutputDims[d] = 1;
63  }
64 
65  // Matrices are in Star,VC format, not default MC,MR
66  delete Zs;
67  delete Ds;
68  delete Ds_Temp;
69  delete Acts;
70  Zs = new StarVCMat(comm->get_model_grid());
71  Ds = new StarVCMat(comm->get_model_grid());
72  Ds_Temp = new StarVCMat(comm->get_model_grid());
73  Acts = new StarVCMat(comm->get_model_grid());
74 
75 
76  // TODO: obtain dimensions from cudnnNet
77  NumNeurons = Channels*OutputDims[0]*OutputDims[1]*OutputDims[2];
78 
79 #ifdef __LIB_CUDNN
80  cudnnLayer = NULL;
81 #endif
82 }

91 void lbann::PoolingLayer::setup(CudnnNet<DataType> *cudnnNet)
92 {
93 #ifdef __LIB_CUDNN
94  // setup cudnn-based pooling layer instance
95  int cudnnSrcTensorDims[] = {1, Channels,
96  InputDims[0], InputDims[1], InputDims[2]};
97  int cudnnDstTensorDims[] = {1, Channels,
98  OutputDims[0], OutputDims[1], OutputDims[2]};
99  if (cudnnLayer) delete cudnnLayer;
100  cudnnLayer = new CudnnLayer<DataType>(cudnnNet);
101  cudnnLayer->setupPoolLayer(PoolDim, cudnnSrcTensorDims, PoolMode,
102  PoolWindowDim, PoolPad, PoolStride,
103  cudnnDstTensorDims);
104  //printf("pooling layer: outputsize: %d\n", cudnnLayer->dstDataSize);
105 
106  // create matrix for output and input deltas
107  Ones(*Zs, cudnnLayer->dstDataSize+1, m_mini_batch_size);
108  Zeros(*Ds, cudnnLayer->srcDataSize+1, m_mini_batch_size);
109  Zeros(*Ds_Temp, cudnnLayer->srcDataSize+1, m_mini_batch_size);
110  Ones(*Acts, cudnnLayer->dstDataSize+1, m_mini_batch_size);
111 
112 #endif
113 }

115 void lbann::PoolingLayer::fp_linearity(ElMat& _WB, ElMat& _X, ElMat& _Z, ElMat& _Y) {
116  // Convert matrices to desired formats
117  DistMatrixReadProxy<DataType,DataType,STAR,VC> XProxy(_X);
118  DistMatrixWriteProxy<DataType,DataType,STAR,VC> ZProxy(_Z);
119  DistMatrixWriteProxy<DataType,DataType,STAR,VC> YProxy(_Y);
120  StarVCMat& X = XProxy.Get();
121  StarVCMat& Z = ZProxy.Get();
122  StarVCMat& Y = YProxy.Get();
123 
124  // Get local matrices
125  Mat& XLocal = X.Matrix();
126  Mat& ZLocal = Z.Matrix();
127  Mat& YLocal = Y.Matrix();
128 
129  for (int j = 0; j < XLocal.Width(); j++) {
130  DataType* src = XLocal.Buffer(0, j);
131  DataType* dst = ZLocal.Buffer(0, j);
132 
133 #ifdef __LIB_CUDNN
134  cudnnLayer->poolForwardHost(src, dst);
135 #endif
136  }
137 
138  // Z and Y are identical after fp linearity step
139  Copy(ZLocal, YLocal);
140 
141 }

146  DistMatrixReadProxy<DataType,DataType,STAR,VC> InputProxy(*fp_input); // TODO: store from fp step

148  StarVCMat& Input = InputProxy.Get();
149 
150  // Get local matrices
151  Mat& InputLocal = Input.Matrix();
152  Mat& OutputLocal = Acts->Matrix();
153  Mat& InputDeltaLocal = Ds_Temp->Matrix();
154  Mat& OutputDeltaLocal = OutputDelta.Matrix();
155 
156  // Iterate through samples in minibatch
157  for (int j = 0; j < InputLocal.Width(); j++) {
158  DataType* src = InputLocal.Buffer(0, j);
159  DataType* dst = OutputLocal.Buffer(0, j);
160  DataType* srcD = InputDeltaLocal.Buffer(0, j);
161  DataType* dstD = OutputDeltaLocal.Buffer(0, j);
162 
163 #ifdef __LIB_CUDNN
164  cudnnLayer->poolBackwardHost(src, dst, dstD, srcD);
