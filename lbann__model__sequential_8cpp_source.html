406 uint lbann::Sequential::add(std::string layerName, int LayerDim,
407  activation_type ActivationType, std::vector<regularizer*> regs)
408 {
409  int prevLayerDim = -1;
410  int layerIndex = Layers.size();
411  int prevLayerIndex = -1;
412  Optimizer *optimizer = optimizer_factory->create_optimizer();
413 
414  if(Layers.size() != 0) {
415  Layer *prev = Layers.back();
416  prevLayerDim = prev->NumNeurons;
417  prevLayerIndex = prev->Index;
418  }
419 
420  if (comm->am_model_master()) {
421  cout << "Adding a layer with input " << prevLayerDim << " and index " << layerIndex << " prev layer index " << prevLayerIndex << endl;
422  }
423 
424  if(layerName.compare("FullyConnected") == 0) {
425  // initalize neural network (layers)
426  Layers.push_back(lfac->create_layer<FullyConnectedLayer>("FullyConnected", layerIndex, prevLayerDim, LayerDim, MiniBatchSize, ActivationType, comm, optimizer, regs));
427  }else if(layerName.compare("SoftMax") == 0) {
428  Layers.push_back(lfac->create_layer<SoftmaxLayer>("SoftMax",layerIndex, prevLayerDim, LayerDim, MiniBatchSize, comm, optimizer));
429  }else {
430  std::cout << "Unknown layer type " << layerName << std::endl;
431  }
432 
433  return layerIndex;
434 }

492 
493 #ifdef __LIB_CUDNN

497  convlayer->setup(&cudnnNet);
