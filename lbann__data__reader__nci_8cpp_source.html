132 bool lbann::data_reader_nci::load(const std::string infile,bool has_header, size_t max_sample_count, bool firstN)
133 {
134  ifstream ifs(infile.c_str());
135  if (!ifs) { std::cout << "\n In load: can't open file : " << infile; exit(1); }
136  m_infile = infile;
137  m_has_header = has_header;
138  string line;
139  int i;
140  std::getline(ifs, line); //skip header
141  while(std::getline (ifs, line) ) {
142  string field;
143  istringstream lstream(line);
144  i=0;
145  m_num_features = 0;
146  //int label;
147  while(getline(lstream, field, ' ')) {
148  i++;
149  if (i > 5) {
150  m_num_features++;
151  }
152  }
153  m_num_samples++;
154  }
155  ifs.close();
156  m_labels.resize(m_num_samples);
157  // reset indices
158  ShuffledIndices.clear();
159  ShuffledIndices.resize(m_num_samples);
160  for (size_t n = 0; n < ShuffledIndices.size(); ++n) {
161  ShuffledIndices[n] = n;
162  }
163 
166  if (max_sample_count != 0) {
167  max_sample_count = __MIN(max_sample_count, m_num_samples);
168  if(!firstN) {
169  std::shuffle(ShuffledIndices.begin(), ShuffledIndices.end(), get_generator());
170  }
171  ShuffledIndices.resize(max_sample_count);
172  if(!firstN) {
173  std::sort(ShuffledIndices.begin(), ShuffledIndices.end());
174  }
175  }
176  return true;
177 }
