49 void lbann::SoftmaxLayer::setup(int numPrevNeurons) {
50  Layer::setup(numPrevNeurons);
51  if(optimizer != NULL) {
52  optimizer->setup(numPrevNeurons+1, NumNeurons);
53  }
54 
55  // Xavier random initialization - Derived from Caffe implementation
56  DataType var_scale = sqrt(3.0 / (numPrevNeurons + 1));
57 
58  if (numPrevNeurons != -1) {
59  // For the softmax layer we do not want to have an extra row propagating the bias term to the output
60  Gaussian(*WB, NumNeurons, numPrevNeurons + 1, (DataType) 0.0, var_scale);
61  if (comm->am_model_master()) {
62  cout << "Softmax Layer " << Index << ": Xavier initialization: input size=" << (numPrevNeurons + 1) << " scale=" << var_scale << " and layer size " << NumNeurons << endl;
63  }
64  Zeros(*WB_D, NumNeurons, numPrevNeurons + 1);
65  Zeros(*Ds, NumNeurons, m_mini_batch_size);
66  Zeros(*Ds_Temp, numPrevNeurons + 1, m_mini_batch_size); // Ds_Temp holds the product of WB^T * Ds
67  Zeros(*Zs, NumNeurons, m_mini_batch_size);
68  }
69  Zeros(*Acts, NumNeurons, m_mini_batch_size);
70  Zeros(Acts_Cost, NumNeurons, m_mini_batch_size);
71 }

77 void lbann::SoftmaxLayer::fp_linearity(ElMat& _WB, ElMat& _X, ElMat& _Z, ElMat& _Y)
78 {
79  // _Z = WB * Xs -- Xs is previous layer Activations
80  // ZsColMax[c,0] = max(_Z[0..numNeurons-1, c]) -- (m_mini_batch_size x 1)
81  // ZsNorm[r,c] = _Z[r,c] - ZsColMax[c,0] -- Column-wise normalized Zs matrix: _Z[r,c] - ZsColMax[1,c]
82  // ZsNormExp[r,c] = exp(ZsNorm[r,c])
83  // ZsNormExpSum[c,0] = sum(ZsNormExp[0..numNeurons-1, c]) -- Column-wise sum over normalized, exponentiated _Z
84  // _Y[r,c] = ZsNormExp[r,c] / ZsNormExpSum[c,0] -- exp(norm(_Z[r,c])) = Sum(exp(norm(Zs[r,c])))
85 
86  // Convert forward prop matrix to MC,MR format
87  // TODO: store this matrix for back prop
88  DistMatrixReadProxy<DataType,DataType,MC,MR> XProxy(_X);
89  DistMat& X = XProxy.Get();
90 
91  // Apply linear transform
92  Gemm(NORMAL, NORMAL, (DataType) 1.0, _WB, X, (DataType) 0.0, _Z);
93 
94  // For each minibatch (column) find the maximimum value
95  Zeros(ZsColMax, m_mini_batch_size, 1);
96  ColumnMax((DistMat&) _Z, ZsColMax);
97 
98  // Redistribute the per-minibatch maximum values
99  Copy(ZsColMax, ZsColMaxStar);
100 
101  // Subtract the max from each column to avoid numerical issues
102  IndexDependentMap(_Z,
103  (std::function<DataType(int,int,DataType)>)([this](int r, int c, DataType z)->
104  DataType{Int rL = this->ZsColMaxStar.LocalRow(c); return z - this->ZsColMaxStar.GetLocal(rL,0);}));
105 
106  // Exponentiation
107  EntrywiseMap(_Z, (std::function<DataType(DataType)>)([](DataType z)->DataType{return exp(z);}));
108 
109  // For each minibatch (column) sum up the exponentiated values
110  Zeros(ZsNormExpSum, m_mini_batch_size, 1);
111  // ColSumMat ZsNormExpSum;
112  ColumnSum((DistMat&) _Z /*ZsNormExp*/, ZsNormExpSum);
113  Copy(ZsNormExpSum, ZsNormExpSumStar);
114 
115  // Divide each entry: exp(x_ij) / Sum_i(exp(x_ij))
116  Copy(_Z /*ZsNormExp*/, _Y);
117  IndexDependentMap(_Y,
118  (std::function<DataType(Int,Int,DataType)>)([this /*ZsNormExpSum*/](Int r, Int c, DataType z)->
119  DataType{Int rL = this->ZsNormExpSumStar.LocalRow(c); return z/this->ZsNormExpSumStar.GetLocal(rL,0);}));
120 
121 #if 1
122  ColSumMat Ycheck(_Y.Grid());
123  Zeros(Ycheck, m_mini_batch_size, 1);
124  ColumnSum((DistMat&) _Y /*ZsNormExp*/, Ycheck);
125  StarMat YcheckStar(_Y.Grid());
126  Copy(Ycheck, YcheckStar);
127  DataType sum = 0.0;
128  for(int i = 0; i < YcheckStar.Height(); i++) {
129  Int l = YcheckStar.LocalRow(i);
130  sum = YcheckStar.GetLocal(l, 0);
131  // sum += YcheckStar.GetLocal(l, 0);
132  // }
133  if(YcheckStar.GetLocal(l, 0) < 0 || (sum >= 1.00001 || sum <= 0.99999)) {
134  if(_Y.Grid().Rank() == 0) {
135  printf("The softmax does not add up %lf\n", sum);
136  }
137  Print(_Y);
138  Print(YcheckStar);
139  }
140  }
141 #endif
142 }

144 void lbann::SoftmaxLayer::bp_linearity()
145 {
146 
147  // Convert forward and backward prop matrices to MC,MR formats
148  DistMatrixReadProxy<DataType,DataType,MC,MR> DsNextProxy(*bp_input);
149  DistMat& DsNext = DsNextProxy.Get();
150  DistMatrixReadProxy<DataType,DataType,MC,MR> XProxy(*fp_input);
151  DistMat& X = XProxy.Get();
152 
153  // delta = (activation - y)
154  // delta_w = delta * activation_prev^T
155  Copy(*Acts, *Ds);
156  Axpy(-1., DsNext, *Ds); // Per-neuron error
157 
158  // Compute the partial delta update for the next lower layer
159  Gemm(TRANSPOSE, NORMAL, (DataType) 1., *WB, *Ds, (DataType) 0., *Ds_Temp);
160 
161  // Compute and display the cost function
162  DistMat Y(Acts->Grid());
163  Copy(DsNext, Y);
164 
165  DataType avg_error = this->computeCost(Y);// + Lambda/2 * WBL2NormSum;
166  // DataType cost = (aggregate_cost / num_backprop_steps);
167  // DataType delta = 0.01 * cost;
168  // if(Output.Grid().Rank() == 0 && (1 || avg_error <= (cost-delta) || avg_error > (cost+delta))) {
169  // cout << "Average of the softmax cost function across the mini-batch " << avg_error << endl;
170  // }
171  aggregate_cost += avg_error;
172  num_backprop_steps++;
173 
174  // by divide mini-batch size
175  Gemm(NORMAL, TRANSPOSE, (DataType) 1.0/get_effective_minibatch_size(), *Ds,
176  X, (DataType) 0., *WB_D);
177 }

179 DataType lbann::SoftmaxLayer::computeCost(DistMat& Y) {
180  // Compute the cost function
181  // cost=-1/m*(sum(sum(groundTruth.*log(a3))))
182  DataType avg_error = 0.0, total_error = 0.0;
183  EntrywiseMap(*Acts, (std::function<DataType(DataType)>)([](DataType z)->DataType{return log(z);}));
184 
185  // DistMat Y(Acts.Grid());
186  // Copy(Output, Y);
187  Hadamard(Y, *Acts, Acts_Cost);
188  ColSumMat MBCost(m_mini_batch_size, 1, Acts->Grid());
189  Zeros(MBCost, m_mini_batch_size, 1);
190  ColumnSum(Acts_Cost, MBCost);
191 
192  int c = 0;
193  // Sum the local, total error
194  for(int r = 0; r < MBCost.LocalHeight(); r++) {
195  total_error += MBCost.GetLocal(r,c);
196  }
197  total_error = mpi::AllReduce(total_error, MBCost.DistComm());
198 
199  avg_error = -1.0 * total_error / m_mini_batch_size;
200  return avg_error;
201 }
