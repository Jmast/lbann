315  template <typename T>
316  void broadcast(T* data, int count, std::vector<int>& dests, int root) {
317  mpi::Group bcast_group;
318  mpi::Comm bcast_comm;
319  std::vector<int> ranks;
320  ranks.push_back(root);
321  ranks.insert(ranks.end(), dests.begin(), dests.end());
322  create_group(ranks, bcast_group);
323  // Elemental doesn't expose this, so we have to reach into its internals.
324  // This lets us create a communicator without involving all of COMM_WORLD.
325  // Use a tag of 0; should not matter unless we're multi-threaded.
326  MPI_Comm_create_group(mpi::COMM_WORLD.comm, bcast_group.group, 0,
327  &(bcast_comm.comm));
328  int translated_root = mpi::Translate(mpi::COMM_WORLD, root, bcast_comm);
329  mpi::Broadcast(data, count, translated_root, bcast_comm);
330  mpi::Free(bcast_comm);
331  mpi::Free(bcast_group);
332  }
