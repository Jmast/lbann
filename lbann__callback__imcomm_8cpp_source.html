47 void lbann_callback_imcomm::setup(Model* m) {
48  if (ct != NONE) {
49  bool add = layer_indices.size() == 0;
50  std::vector<Layer*>& layers = m->get_layers();
51  for (Layer* layer : layers) {
52  uint idx = layer->get_index();
53  if (add || layer_indices.find(idx) != layer_indices.end()) {
54  // Ensure index is present (overwrites if already there).
55  layer_indices.insert(idx);
56  // Skip adding matrices when we don't need to.
57  if (!ct_does_quantization()) continue;
58  // TODO: handle case where WB_D is in other matrix distribution
59  DistMat& WB_D = (DistMat&) layer->get_weights_biases_gradient();
60  quantization_errors.emplace(idx, Mat{});
61  Zeros(quantization_errors[idx], WB_D.LocalHeight(), WB_D.LocalWidth());
62  im_quantization_errors.emplace(idx, Mat{});
63  if (ct == ONEBIT_QUANTIZATION) {
64  // Set up gradient history and SGD optimizer for one-bit quantization.
65  gradhistories.emplace(idx, Mat{});
66  if (layer->optimizer != nullptr) {
67  // TODO: This leaks the old optimizer.
68  layer->optimizer = new SGD<DistMat>(
69  layer->comm, layer->optimizer->get_learning_rate(),
70  0.0f, 0.0f, false);
71  layer->optimizer->setup(layer->WB->Width(), layer->WB->Height());
72  }
73  }
74  }
75  }
76  }

79 void lbann_callback_imcomm::on_epoch_end(Model* m) {
80  lbann_comm* comm = m->get_comm();
81  if (comm->get_num_models() == 1) {
82  return; // No point with only one model.
83  }
84  if (ct_does_quantization()) {
85  std::vector<Layer*>& layers = m->get_layers();
86  for (size_t l = 0; l < layers.size(); ++l) {
87  if (layer_indices.find(layers[l]->get_index()) == layer_indices.end()) {
88  continue;
89  }
90  comm->intermodel_sum_matrix(quantization_errors[l]);
91  // TODO: handle case where WB_D is in other matrix distribution
92  DistMat& WB_D = (DistMat&) layers[l]->get_weights_biases_gradient();
93  Mat& local_mat = WB_D.Matrix();
94  local_mat = quantization_errors[l];
95  // Apply optimizer update again.
96  layers[l]->update();
97  Zeros(quantization_errors[l], quantization_errors[l].Height(),
98  quantization_errors[l].Width());
99  }
100  }
101 }

103 void lbann_callback_imcomm::on_backward_prop_end(Model* m) {
104  lbann_comm* comm = m->get_comm();
105  if (comm->get_num_models() == 1) {
106  return; // No point with only one model.
107  }
108  std::vector<Layer*>& layers = m->get_layers();
109  for (size_t l = 0; l < layers.size(); ++l) {
110  if (layer_indices.find(layers[l]->get_index()) == layer_indices.end()) {
111  continue;
112  }
113  double start_time = get_time();
114  // TODO: handle case where WB_D is in other matrix distribution
115  DistMat& WB_D = (DistMat&) layers[l]->get_weights_biases_gradient();
116  switch (ct) {
117  case NONE:
118  break;
119  case NORMAL:
120  comm->intermodel_sum_matrix(WB_D);
121  break;
122  case ONEBIT_QUANTIZATION:
123  quantizer.intermodel_sum_quantized(
124  comm, WB_D, quantization_errors[l], im_quantization_errors[l], true,
125  &(gradhistories[l]));
126  break;
127  case THRESH_QUANTIZATION:
128  // TODO: Don't hardcode thresholds.
129  quantizer.intermodel_sum_threshold_quantized(
130  comm, WB_D, quantization_errors[l], 1.0f, -1.0f,
131  im_quantization_errors[l], false);
132  break;
133  case COMPRESSED_THRESH_QUANTIZATION:
134  // TODO: Don't hardcode thresholds.
135  quantizer.intermodel_sum_threshold_quantized(
136  comm, WB_D, quantization_errors[l], 1.0f, -1.0f,
137  im_quantization_errors[l], true);
138  break;
139  case ADAPTIVE_THRESH_QUANTIZATION:
140  // TODO: Don't hardcode proportion.
141  quantizer.intermodel_sum_adaptive_threshold_quantized(
142  comm, WB_D, quantization_errors[l], 5,
143  im_quantization_errors[l], false);
144  break;
145  case COMPRESSED_ADAPTIVE_THRESH_QUANTIZATION:
146  // TODO: Don't hardcode proportion.
147  quantizer.intermodel_sum_adaptive_threshold_quantized(
148  comm, WB_D, quantization_errors[l], 5,
149  im_quantization_errors[l], true);
150  break;
151  }
152  double im_time = get_time() - start_time;
153  if (summarizer != nullptr && ct != NONE) {
154  summarizer->reduce_scalar(
155  "layer" + std::to_string(
156  static_cast<long long>(layers[l]->get_index())) +
157  "/imcomm_time",
158  im_time, m->get_cur_step());
159  }
160  }
161 }
