39 // FullyConnectedLayer : single network layer class
41 // WB structure: (num units "neurons / filters" x (num features + 1))
42 // Each row represents a neuron / filter
43 // There is a column for each feature coming in from the previous layer plus 1 for the bias
44 // [W0 ... B0]
45 // [| |]
46 // [Wn Bn]
47 // [0 ... 0 1] - Initialize the final row to be all zeros and 1 in the bias to properly
48 // set the bias for the next layer
49 // WB_D structure:
50 // [dW dB]
51 // [0 ... 0 0]
52 // D structure:
53 // [D ]
54 // [0 ... 0 0]
55 // Z, Zs, Act, Acts structure:
56 // [Acts ]
57 // [1 ... 1 1]

86 void lbann::FullyConnectedLayer::setup(int numPrevNeurons) {
87  Layer::setup(numPrevNeurons);
88  if(optimizer != NULL) {
89  optimizer->setup(numPrevNeurons+1, NumNeurons+1);
90  }
91 
92  // Xavier random initialization - Derived from Caffe implementation
93  DataType var_scale = sqrt(3.0 / (numPrevNeurons + 1));
94 
95  if (numPrevNeurons != -1) {
96  Gaussian(*WB, NumNeurons + 1, numPrevNeurons + 1, (DataType) 0.0, var_scale); // use var_scale, instead of 0.1
97  if (comm->am_model_master()) {
98  cout << "Fully Connected Layer " << Index << ": Xavier initialization: input size=" << (numPrevNeurons + 1) << " scale=" << var_scale << " and layer size " << NumNeurons << endl;
99  }
100  // Set the last row to all zeros and a 1 in the last column to set the bias term for
101  // activation layer: 0 0 0 0 ... 0 1
102  Int ngrows = WB->Height();
103  Int ngcols = WB->Width();
104  Int nrows = WB->LocalHeight();
105  Int ncols = WB->LocalWidth();
106  Int r = nrows-1;
107  Int gr = WB->GlobalRow(r);
108  Int gc = WB->GlobalCol(ncols-1);
109  if(gr == ngrows - 1) { // Bias initialization row
110  for (int c = 0; c < ncols; c++) {
111  WB->SetLocal(r, c, 0.0); // Set the bias row back to 0.0
112  }
113  if(gc == ngcols-1) {
114  WB->SetLocal(r, ncols-1, 1.0); // and 1.0
115  }
116  }
117  Zeros(*WB_D, NumNeurons + 1, numPrevNeurons + 1);
118  Zeros(*Ds, NumNeurons + 1, m_mini_batch_size);
119  Zeros(*Ds_Temp, numPrevNeurons + 1, m_mini_batch_size); // Ds_Temp holds the product of WB^T * Ds
120  Zeros(*Zs, NumNeurons + 1, m_mini_batch_size);
121  View(WB_view, *WB, IR(0, WB->Height() - 1), IR(0, WB->Width()));
122  View(WB_D_view, *WB_D, IR(0, WB_D->Height() - 1), IR(0, WB_D->Width()));
123  }
124  Zeros(*Acts, NumNeurons + 1, m_mini_batch_size);
125 
126 #if 0
127  printf("Layer[%d] has %d neurons and %d inputs\n", Index, NumNeurons + 1, numPrevNeurons + 1);
128  printf("trainMB have allocated Layers[%d]->Acts %d entries (%d x %d)\n", Index, Acts->AllocatedMemory(), Acts->Height(), Acts->Width());
129  printf("trainMB have allocated Layers[%d]->WB %d entries (%d x %d)\n", Index, WB->AllocatedMemory(), WB->Height(), WB->Width());
130  printf("trainMB have allocated Layers[%d]->WB_D %d entries (%d x %d)\n", Index, WB_D->AllocatedMemory(), WB_D->Height(), WB_D->Width());
131  printf("trainMB have allocated Layers[%d]->Ds %d entries (%d x %d)\n", Index, Ds->AllocatedMemory(), Ds->Height(), Ds->Width());
132  printf("trainMB have allocated Layers[%d]->Ds_Temp %d entries (%d x %d)\n", Index, Ds_Temp.AllocatedMemory(), Ds_Temp.Height(), Ds_Temp.Width());
133 #endif
134 }

136 void lbann::FullyConnectedLayer::fp_linearity(ElMat& _WB, ElMat& _X, ElMat& _Z, ElMat& _Y)
137 {
138  // Convert matrices to desired format
139  DistMatrixReadProxy<DataType,DataType,MC,MR> WBProxy(_WB);
140  DistMatrixReadProxy<DataType,DataType,MC,MR> XProxy(_X); // TODO: Store for bp step
141  DistMatrixWriteProxy<DataType,DataType,MC,MR> ZProxy(_Z);
142  DistMatrixWriteProxy<DataType,DataType,MC,MR> YProxy(_Y);
143  DistMat& WB = WBProxy.Get();
144  DistMat& X = XProxy.Get();
145  DistMat& Z = ZProxy.Get();
146  DistMat& Y = YProxy.Get();
147 
148  // Apply forward prop linearity
149  Gemm(NORMAL, NORMAL, (DataType) 1., WB, X, (DataType) 0., Z);
150  Copy(Z, Y);
151 }

153 void lbann::FullyConnectedLayer::bp_linearity()
154 {
155  // Convert forward and backward prop matrices to MC,MR format
156  DistMatrixReadProxy<DataType,DataType,MC,MR> DsNextProxy(*bp_input);
157  DistMatrixReadProxy<DataType,DataType,MC,MR> XProxy(*fp_input); // TODO: store from fp step
158  DistMat& DsNext = DsNextProxy.Get();
159  DistMat& X = XProxy.Get();
160 
161  // Compute the delta using the results from "next" deeper layer
162  Hadamard(DsNext, *Zs, *Ds);
163  // Compute the partial delta update for the next lower layer
164  Gemm(TRANSPOSE, NORMAL, (DataType) 1., *WB, DsNext, (DataType) 0., *Ds_Temp);
165  // BVE - an alternative approach is to compute the mean
166  Gemm(NORMAL, TRANSPOSE, (DataType) 1.0/get_effective_minibatch_size(), *Ds,
167  X, (DataType) 0., *WB_D);
168 }

170 DataType lbann::FullyConnectedLayer::computeCost(DistMat &deltas) {
171  DataType avg_error = 0.0, total_error = 0.0;
172  // Compute the L2 norm on the deltas (activation - y)
173  ColSumMat norms;
174  ColumnTwoNorms(deltas, norms);
175  int c = 0;
176  // Sum the local, total error
177  for(int r = 0; r < norms.LocalHeight(); r++) {
178  total_error += norms.GetLocal(r,c);
179  }
180  mpi::AllReduce(total_error, norms.DistComm());
181  avg_error = total_error / norms.Height();
182  return avg_error;
183 }

200 DataType lbann::FullyConnectedLayer::checkGradient(Layer& PrevLayer, const DataType Epsilon)
201 {
202  DistMat WB_E1(WB->Grid());
203  DistMat WB_E2(WB->Grid());
204  DistMat Zs_E1(Zs->Grid());
205  DistMat Zs_E2(Zs->Grid());
206  DistMat Acts_E1(Acts->Grid());
207  DistMat Acts_E2(Acts->Grid());
208  DataType grad_diff = 0;
209  DataType grad_sum = 0;
210 
211  Zeros(WB_E1, WB->Height(), WB->Width());
212  Zeros(WB_E2, WB->Height(), WB->Width());
213  Zeros(Zs_E1, Zs->Height(), Zs->Width());
214  Zeros(Zs_E2, Zs->Height(), Zs->Width());
215  Zeros(Acts_E1, Acts->Height(), Acts->Width());
216  Zeros(Acts_E2, Acts->Height(), Acts->Width());
217 
218  Copy(*WB, WB_E1);
219  Copy(*WB, WB_E2);
220 
221  DataType sum_error = 0;
222  int prow = 0;
223  int pcol = 0;
224 
225  if(WB_E1.IsLocal(prow, pcol)) {
226  int _prow = WB_E1.LocalRow(prow);
227  int _pcol = WB_E1.LocalCol(pcol);
228  WB_E1.SetLocal(_prow, _pcol, WB_E1.GetLocal(_prow, _pcol) + Epsilon);
229  }
230  if(WB_E2.IsLocal(prow, pcol)) {
231  int _prow = WB_E2.LocalRow(prow);
232  int _pcol = WB_E2.LocalCol(pcol);
233  WB_E2.SetLocal(_prow, _pcol, WB_E2.GetLocal(_prow, _pcol) - Epsilon);
234  }
235  for (int row = 0; row < WB->Height(); row++) {
236  for (int col = 0; col < WB->Width(); col++) {
237  // printf("Updating %d: %d x %d\n", Index, row, col);
238  if(WB_E1.IsLocal(prow, pcol)) {
239  int _prow = WB_E1.LocalRow(prow);
240  int _pcol = WB_E1.LocalCol(pcol);
241  WB_E1.SetLocal(_prow, _pcol, WB_E1.GetLocal(_prow, _pcol) - Epsilon);
242  }
243  if(WB_E2.IsLocal(prow, pcol)) {
244  int _prow = WB_E2.LocalRow(prow);
245  int _pcol = WB_E2.LocalCol(pcol);
246  WB_E2.SetLocal(_prow, _pcol, WB_E2.GetLocal(_prow, _pcol) + Epsilon);
247  }
248  if(WB_E1.IsLocal(row, col)) {
249  int _row = WB_E1.LocalRow(row);
250  int _col = WB_E1.LocalCol(col);
251  WB_E1.SetLocal(_row, _col, WB_E1.GetLocal(_row, _col) + Epsilon);
252  }
253  if(WB_E2.IsLocal(row, col)) {
254  int _row = WB_E2.LocalRow(row);
255  int _col = WB_E2.LocalCol(col);
256  WB_E2.SetLocal(_row, _col, WB_E2.GetLocal(_row, _col) - Epsilon);
257  }
258 
259  // J(theta)
260  this->fp_linearity(*WB, *(PrevLayer.Acts), *Zs, *Acts);
261  this->fp_nonlinearity();
262 
263  // J(thetaPlus(i))
264  this->fp_linearity(WB_E1, *(PrevLayer.Acts), Zs_E1, Acts_E1);
265  this->fp_nonlinearity();
266 
267  // J(thetaMinus(i))
268  this->fp_linearity(WB_E2, *(PrevLayer.Acts), Zs_E2, Acts_E2);
269  this->fp_nonlinearity();
270 
271  // this->getCost();
272 
273  // gradApprox(i) = J(thetaPlus(i)) - J(thetaMinus(i)) / (2*Epsilon)
274  Axpy(-1.0, *Acts, Acts_E1);
275  Axpy(-1.0, *Acts, Acts_E2);
276 
277  bool bad_E1 = false, bad_E2 = false;
278  for(int r = 0; r < Acts->Height(); r++) {
279  for(int c = 0; c < Acts->Width(); c++) {
280  if(Acts_E1.IsLocal(r,c)) {
281  int _r = Acts_E1.LocalRow(r);
282  int _c = Acts_E1.LocalCol(c);
283  if((Acts_E1.GetLocal(_r,_c) > 1e-12 || Acts_E1.GetLocal(_r,_c) < -1e-12) && r != row) {
284  bad_E1 = true;
285  // cout << "Acts_E1=["<< r << "," << c << "]="<<Acts_E1.GetLocal(_r,_c)<<endl;
286  }
287  }
288  if(Acts_E2.IsLocal(r,c)) {
289  int _r = Acts_E2.LocalRow(r);
290  int _c = Acts_E2.LocalCol(c);
291  if((Acts_E2.GetLocal(_r,_c) > 1e-12 || Acts_E2.GetLocal(_r,_c) < -1e-12) && r != row) {
292  bad_E2 = true;
293  // cout << "Acts_E2=["<< r << "," << c << "]="<<Acts_E2.GetLocal(_r,_c)<<endl;
294  }
295  }
296  }
297  }
298 
299  if(bad_E1) {
300  if(Acts_E1.Grid().Rank() == 0) {
301  printf("BAD ENTRY Acts_E1 %d x %d\n", row, col);
302  }
303  cout.precision(20);
304  Print(Acts_E1);
305  if(Acts_E1.Grid().Rank() == 0) {
306  printf("Acts\n");
307  }
308  Print(*Acts);
309  }
310  if(bad_E2) {
311  if(Acts_E2.Grid().Rank() == 0) {
312  printf("BAD ENTRY Acts_E2 %d x %d\n", row, col);
313  }
314  Print(Acts_E2);
315  }
316 
317  prow = row;
318  pcol = col;
319  }
320  }
321  DataType grad_error = sqrt(grad_diff / grad_sum);
322  return grad_error;
323 }
