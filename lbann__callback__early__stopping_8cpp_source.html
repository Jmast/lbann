36 void lbann_callback_early_stopping::on_test_end(Model* m) {
38  //DataType acc = m->get_validation_accuracy();
39  // May need to change this callback to on_train_end or something that runs
40  // after validation.
41  DataType acc = m->get_test_accuracy();
42  if (acc > last_acc) {
43  last_acc = acc;
44  wait = 0;
45  } else {
46  if (wait >= patience) {
47  m->set_terminate_training(true);
48  if (m->get_comm()->am_model_master()) {
49  std::cout << "Model " << m->get_comm()->get_model_rank() <<
50  " terminating training due to early stopping" << std::endl;
51  }
52  } else {
53  ++wait;
54  }
55  }
56 }
