100 void lbann::Dnn::train(int NumEpoch, bool EvaluateEveryEpoch)
101 {
102  do_train_begin_cbs();
103  // do main loop for epoch
104  for (int epoch = 0; epoch < NumEpoch; epoch++) {
105  if (get_terminate_training()) {
106  // Training has been terminated for this model.
107  break;
108  }
109  ++cur_epoch;
110  do_epoch_begin_cbs();
113  for (size_t l = 0; l < Layers.size(); l++) {
114  Layers[l]->m_execution_mode = training;
115  }
116 
117  long num_samples = 0;
118  long num_errors = 0;
119 
120  // Trigger a shuffle of the input data
121  bool epoch_done;
122  do {
123  epoch_done = trainBatch(&num_samples, &num_errors);
124  }while(!epoch_done);
125 
127  training_accuracy = (float)(num_samples - num_errors) / num_samples * 100.0f;
128 
129  if(EvaluateEveryEpoch) {
130  do_validation_begin_cbs();
131  validation_accuracy = evaluate();
132  do_validation_end_cbs();
133  }
134 
135  do_epoch_end_cbs();
136  }
137  do_train_end_cbs();
138 }

146 bool lbann::Dnn::trainBatch(long *num_samples, long *num_errors)
147 {
148  do_batch_begin_cbs();
149  bool data_set_processed = false;
150 
151  do_model_forward_prop_begin_cbs();
152  // forward propagation (mini-batch)
153  DataType L2NormSum = 0;
154  for (size_t l = 0; l < Layers.size(); l++) {
155  do_layer_forward_prop_begin_cbs(Layers[l]);
156  L2NormSum = Layers[l]->forwardProp(L2NormSum);
157  do_layer_forward_prop_end_cbs(Layers[l]);
158  }
159  *num_errors += (long) L2NormSum;
160  *num_samples += MiniBatchSize;
161  do_model_forward_prop_end_cbs();
162 
163  // Update training accuracy.
164  training_accuracy = ((DataType) *num_errors) / ((DataType) *num_samples) * 100.0;
165  ++cur_step;
166 
167  do_model_backward_prop_begin_cbs();
168  // backward propagation (mini-batch)
169  for (size_t l = Layers.size(); l-- > 0;) {
170  do_layer_backward_prop_begin_cbs(Layers[l]);
171  Layers[l]->backProp();
172  do_layer_backward_prop_end_cbs(Layers[l]);
173  }
174  do_model_backward_prop_end_cbs();
175 
176  // Make sure that the output error is periodically reported in backpropagation
177 
180  // for (size_t l = 0; l < Layers.size(); l++) {
181  for (size_t l = Layers.size(); l-- > 0;) {
182  data_set_processed = Layers[l]->update();
183  }
184 
185  do_batch_end_cbs();
186 
187  return data_set_processed;
188 }
