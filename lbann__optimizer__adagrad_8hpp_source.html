81  void setup(int input_dim, int num_neurons) {
82  if (comm->am_model_master()) {
83  printf("Setting up Adagrad optimizer with cache size %d x %d\n", num_neurons, input_dim);
84  }
85  Zeros(WB_D_Cache, num_neurons, input_dim);

88  if (comm->am_model_master()) {

93  void update_weight_bias_matrix(ElMat& WB_D, ElMat& WB) {
94  Copy(WB_D, WB_D_Temp);
95  // Square each entry of the WB_D matrix
96  EntrywiseMap(WB_D_Temp, std::function<DataType(DataType)>(_sq));
97  // Add squared value to WB_D_Cache
98  Axpy(1., WB_D_Temp, WB_D_Cache);
99 
100  Copy(WB_D_Cache, WB_D_Temp);
101  // Compute the inverse of the square root of the historical gradient (with a small perturbation)
102  EntrywiseMap(WB_D_Temp, std::function<DataType(DataType)>(_sqrt));
103  Copy(WB_D, WB_D_Temp2);
104  Hadamard(WB_D_Temp2, WB_D_Temp, WB_D);
105 
106  // WBL2NormSum = 0.0;
107  Axpy((DataType)-lr, WB_D, WB);
108  }

142  bool loadFromCheckpointShared(const char* dir, int Index, uint64_t* bytes) {
143  int rank = WB_D_Cache.Grid().Rank();
144 
145  char path[512];
146  struct stat buffer;
147 
148  // read in the cache of gradients for WB
149  sprintf(path, "%s/WB_D_CACHE_L%d_%03dx%03d.bin", dir, Index, WB_D_Cache.Height()-1, WB_D_Cache.Width()-1);
150 
151  // check whether file exists
152  int exists = 0;
153  if (rank == 0 && stat(path, &buffer) == 0) {
154  exists = 1;
155  }
156  MPI_Bcast(&exists, 1, MPI_INT, 0, MPI_COMM_WORLD);
157 
158  // read WB_D_Cache file
159  if (rank == 0) {
160  cout << "Restoring layer " << Index << " from file " << path << endl;
161  }
162  Read(WB_D_Cache, path, BINARY, 1);
163  //Read_MPI(WB_D_Cache, path, BINARY, 1);
164 
165  if (rank == 0) {
166  *bytes += 2 * sizeof(int) + WB_D_Cache.Height() * WB_D_Cache.Width() * sizeof(DataType);
167  }
168 
169  return true;
170  }
