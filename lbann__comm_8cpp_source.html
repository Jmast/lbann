38 lbann::lbann_comm::lbann_comm(int _procs_per_model) :
39  procs_per_model(_procs_per_model), num_model_barriers(0),
40  num_intermodel_barriers(0), num_global_barriers(0), bytes_sent(0),
41  bytes_received(0) {
42  if (procs_per_model == 0) {
43  procs_per_model = mpi::Size(mpi::COMM_WORLD);
44  }
45  num_models = mpi::Size(mpi::COMM_WORLD) / procs_per_model;
46  model_rank = mpi::Rank(mpi::COMM_WORLD) / procs_per_model;
47  rank_in_model = mpi::Rank(mpi::COMM_WORLD) % procs_per_model;
48  mpi::Split(mpi::COMM_WORLD, model_rank, rank_in_model, model_comm);
49  mpi::Split(mpi::COMM_WORLD, rank_in_model, model_rank, intermodel_comm);
50  grid = new Grid(model_comm);
51 }
52 
53 lbann::lbann_comm::~lbann_comm() {
54  delete grid;
55  mpi::Free(model_comm);
56  mpi::Free(intermodel_comm);
57 }
