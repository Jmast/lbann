74 DataType lbann::target_layer_distributed_minibatch::forwardProp(DataType prev_WBL2NormSum) {
75  DataReader *data_reader = target_layer::select_data_reader();
76 
77  if (comm->get_rank_in_model() == m_root) {
78  Zero(Y_local);

82  if (comm->get_rank_in_model() == m_root) {
83  CopyFromRoot(Y_local, Ys);
84  }else {
85  CopyFromNonRoot(Ys);

88  comm->model_barrier();
89 
91  int num_errors = 0;
94  for (int n = 0; n < Y_local.Width(); n++) {
95  int targetidx = -1;
96  float targetmax = 0;
97  for (int m = 0; m < Y_local.Height(); m++) {
98  if (targetmax < Y_local.Get(m, n)) {
99  targetmax = Y_local.Get(m, n);
100  targetidx = m;
101  }
102  }
103  int labelidx = -1;
104  float labelmax = 0;
105  for (int m = 0; m < fp_input->Height(); m++) {
106  if (labelmax < fp_input->Get(m, n)) {
107  labelmax = fp_input->Get(m, n);
108  labelidx = m;
109  }
110  }
111  if (targetidx != labelidx)
112  num_errors++;
113 
114  }
115 
116  return num_errors;
