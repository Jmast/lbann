98 void lbann::NetworkParams::parse_network_string(void) {
99  Network.clear();
100  const std::string delim = ",";
101  size_t start = 0U;
102  size_t end = NetworkStr.find(delim);
103  while(end != std::string::npos) {
104  Network.push_back(std::stoi(NetworkStr.substr(start, end - start), nullptr, 0));
105  start = end + delim.length();
106  end = NetworkStr.find(delim, start);
107  }
108  Network.push_back(std::stoi(NetworkStr.substr(start, end), nullptr, 0));
109 }
